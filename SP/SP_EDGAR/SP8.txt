  `id` int(11) NOT NULL AUTO_INCREMENT,
  `codsalon` int(11) DEFAULT NULL,
  `codtipometa` int(11) DEFAULT NULL,
  `meta` int(11) DEFAULT NULL,
  `fecha_registro` timestamp NULL DEFAULT NULL,
  `fecha_mod` timestamp NULL DEFAULT NULL,
  `usuario_modifico` varchar(100) DEFAULT NULL,
