CREATE DEFINER=`developer2021`@`%` PROCEDURE `SP_SALONES_META`(
_codciclo INT,
_codlinea INT
)
BEGIN
SELECT  
a.codsalones,a.nomsalones,a.codciclo,b.meta 
FROM salones a
LEFT JOIN meta b on a.codsalones = b.codsalon
LEFT JOIN ccostos c on c.codccostos= a.codccosto
LEFT JOIN lineas d on d.codlinea=c.codlinea

WHERE
a.codciclo=_codciclo AND d.codlinea=_codlinea;
END
