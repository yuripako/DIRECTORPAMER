CREATE DEFINER=`developer2021`@`%` PROCEDURE `SP_DIRECTOR_xxx_CRUD_META`(
_meta INT,
_codsalon INT
)
BEGIN

DECLARE count INT;
SELECT count(*) INTO count FROM meta WHERE codsalon=_codsalon;
IF count > 0 then
UPDATE meta SET codsalon=_codsalon, meta=_meta
 WHERE codsalon=_codsalon;
 
ELSE 

INSERT INTO meta(codsalon,meta,fecha_registro,estado) VALUES(_codsalon,_meta,NOW(),1);
	   
END IF;

END
