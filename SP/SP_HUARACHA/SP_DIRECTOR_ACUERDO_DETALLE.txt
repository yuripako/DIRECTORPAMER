CREATE PROCEDURE pamervirtual.SP_DIRECTOR_ACUERDO_DETALLE(
	intAcuerdo int
)
BEGIN
	SET lc_time_names = 'es_ES';
	SET session group_concat_max_len=1000000;
 	
	select 
	acuerdo_id, 
	acuerdo_tipo, 
	DATE_FORMAT(creado,'%d %b. %Y'),
	titulo, 
	resumen, 
	acuerdo,
	multimedias, 
	calificacion 
	from acuerdos_orientacion 
	where acuerdo_id = intAcuerdo;
END
